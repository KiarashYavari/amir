46 class IndentedHelpFormatterWithNL(IndentedHelpFormatter):
47  def format_option(self, option):
48  result = []
49  opts = self.option_strings[option]
50  opt_width = self.help_position - self.current_indent - 2
51  if len(opts) > opt_width:
52  opts = "%*s%s\n" % (self.current_indent, "", opts)
53  indent_first = self.help_position
54  else: # start help on same line as opts
55  opts = "%*s%-*s " % (self.current_indent, "", opt_width, opts)
56  indent_first = 0
57  result.append(opts)
58  if option.help:
59  #help_text = option.help
60  help_text = self.expand_default(option)
61  help_lines = []
62  #help_text = "\n".join([x.strip() for x in help_text.split("\n")])
63  for para in help_text.split("\n\n"):
64  help_lines.extend(textwrap.wrap(para, self.help_width))
65  # if len(help_lines):
66  # help_lines[-1] += "\n"
67  result.append("%*s%s\n" % (indent_first, "", help_lines[0]))
68  result.extend(["%*s%s\n" % (self.help_position, "", line)
69  for line in help_lines[1:]])
70  elif opts[-1] != "\n":
71  result.append("\n")
72  return "".join(result)
